type Verdict = "APPLY" | "REVIEW" | "PASS"

function getVerdict(r: AnalysisResult): Verdict {
    const comp = r.competitiveness
    const level = r.role_level
    const hiddenCount = r.hidden_requirements?.length ?? 0
    if (comp === "Very High" && hiddenCount >= 5) return "PASS"
    if (
        (comp === "Low" || comp === "Medium") &&
        (level === "Entry" || level === "Early Career")
    )
        return "APPLY"
    return "REVIEW"
}

function getGenericScore(r: AnalysisResult): number {
    const compMap: Record<string, number> = {
        Low: 75,
        Medium: 55,
        High: 40,
        "Very High": 25,
    }
    let base = compMap[r.competitiveness] ?? 50
    const levelAdj: Record<string, number> = {
        Entry: -5,
        "Early Career": 0,
        "Mid-Level": 5,
        Experienced: 10,
    }
    base += levelAdj[r.role_level] ?? 0
    return Math.max(10, Math.min(90, base))
}

function getCompSegments(comp: string): number {
    const map: Record<string, number> = {
        Low: 1,
        Medium: 2,
        High: 3,
        "Very High": 4,
    }
    return map[comp] ?? 2
}

        const sessionId = existing || crypto.randomUUID()
        if (!existing) localStorage.setItem("jobfit_session_id", sessionId)

        const src =
            url.searchParams.get("utm_source") ||
            sessionStorage.getItem("signal_utm_source") ||
            null
        const med =
            url.searchParams.get("utm_medium") ||
            sessionStorage.getItem("signal_utm_medium") ||
            null
        const camp =
            url.searchParams.get("utm_campaign") ||
            sessionStorage.getItem("signal_utm_campaign") ||
            null

            if (src) sessionStorage.setItem("signal_utm_source", src)
            if (med) sessionStorage.setItem("signal_utm_medium", med)
            if (camp) sessionStorage.setItem("signal_utm_campaign", camp)

        fetch(API_BASE + "/api/track", {
            method: "POST",
            headers: { "Content-Type": "application/json" },

                session_id: sessionId,
                page_path: window.location.pathname || "/analyze",
                page_name: "job_analysis_page",
                referrer: document.referrer || null,

                utm_medium: med,
                utm_campaign: camp,

    }, [])

    // Rotate loading messages
    React.useEffect(() => {
        if (!loading) return
        let idx = 0
        const iv = setInterval(() => {
            idx = (idx + 1) % LOADING_MSGS.length
            setLoadingMsg(LOADING_MSGS[idx])
        }, 2500)
        return () => clearInterval(iv)
    }, [loading])

    async function handleAnalyze() {
        setError("")
        const trimmed = jd.trim()
        if (trimmed.length < 100) {
            setError(
                "Paste the full job description. It needs to be at least 100 characters."
            )
            return

        setLoading(true)
        setLoadingMsg(LOADING_MSGS[0])
        try {
            const sessionId =
                localStorage.getItem("jobfit_session_id") || undefined
            const src = sessionStorage.getItem("signal_utm_source") || undefined
            const med = sessionStorage.getItem("signal_utm_medium") || undefined
            const camp =
                sessionStorage.getItem("signal_utm_campaign") || undefined

            const res = await fetch(API_BASE + "/api/job-analysis", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    job_description: trimmed,
                    // User-provided labels — API treats these as optional
                    // overrides and only uses them when non-empty.
                    company_name: companyName.trim() || undefined,
                    job_title: jobTitle.trim() || undefined,
                    session_id: sessionId,
                    utm_source: src,
                    utm_medium: med,
                    utm_campaign: camp,
                }),
            })
            const data = await res.json()
            if (!res.ok)
                throw new Error(data?.error || "Analysis failed. Try again.")
            setResult(data as AnalysisResult)
        } catch (err: any) {
            setError(err?.message || "Something went wrong. Try again.")
        } finally {
            setLoading(false)
        }
    }

    function handleCTA() {
        try {
            sessionStorage.setItem(
                "signal_jd",
                JSON.stringify({
                    job_description: jd.trim(),
                    analysis: result,
                    company_name: result?.company_name,
                    timestamp: Date.now(),
                })
            )
        } catch {}

        // Track upgrade CTA click (analyzer results -> GHL purchase).
        // Uses sendBeacon so the event survives the redirect. We send as
        // text/plain (not application/json) because application/json triggers
        // a CORS preflight that sendBeacon cannot wait for during page unload,
        // so Chrome silently drops the beacon. The server parses the body via
        // req.json() which ignores Content-Type, so this is safe.
        const sessionId = localStorage.getItem("jobfit_session_id") || undefined
        const src = sessionStorage.getItem("signal_utm_source") || undefined
        const payload = JSON.stringify({
            page_name: "upgrade_cta_click",
            session_id: sessionId,
            page_path: window.location.pathname,
            utm_source: src,
        })
        try {
            if (typeof navigator !== "undefined" && navigator.sendBeacon) {
                const blob = new Blob([payload], { type: "text/plain" })
                navigator.sendBeacon(API_BASE + "/api/track", blob)
            } else {
                fetch(API_BASE + "/api/track", {
                    method: "POST",
                    headers: { "Content-Type": "text/plain" },
                    body: payload,
                    keepalive: true,
                }).catch(() => {})
            }
        } catch {}

        // Small delay lets the beacon queue before navigation. sendBeacon is
        // supposed to survive unload, but in practice ~50ms makes delivery
        // much more reliable across browsers.
        setTimeout(() => {
            window.location.assign(GHL_URL)
        }, 50)
    }

        setJd("")
        setCompanyName("")
        setJobTitle("")

                ready: false,
            }

    }

    return (
        <div style={pageStyle}>
            {/* Injected keyframes */}
            <style

                    ...orbBase,
                    width: 500,
                    height: 500,
                    background: "rgba(81,173,229,0.1)",
                    top: -150,
                    right: -100,
                }}
            />

                style={{
                    ...orbBase,
                    width: 400,
                    height: 400,
                    background: "rgba(254,176,106,0.07)",
                    top: 300,
                    left: -120,
                }}
            />
            <div
                style={{
                    ...orbBase,
                    width: 300,
                    height: 300,
                    background: "rgba(192,132,252,0.07)",
                    bottom: 100,
                    right: -60,
                }}
            />
            {result ? (
                <div
                    style={{
                        ...orbBase,
                        width: 280,
                        height: 280,
                        background: "rgba(74,222,128,0.06)",
                        top: 900,
                        left: -40,
                    }}
                />
            ) : null}

            <div style={{ position: "relative" as const, zIndex: 1 }}>
                {/* Topbar */}
                <div style={topbar}>
                    <div>
                        <div
                            style={{
                                fontSize: 17,
                                fontWeight: 900,
                                letterSpacing: "-.02em",
                            }}
                        >
                            SIGNAL
                            <span style={{ color: C.orange }}>{"\u2014"}</span>
                        </div>
                        <div
                            style={{
                                fontSize: 9,
                                color: C.blue,
                                letterSpacing: ".1em",
                                fontWeight: 600,
                                marginTop: 1,
                                fontFamily: MONO,
                            }}
                        >
                            BY WORKFORCE READY NOW
                        </div>
                    </div>
                    <div
                        style={{
                            fontSize: 11,
                            color: result ? "#4a6a85" : C.green,
                            display: "flex",
                            alignItems: "center",
                            gap: 6,
                        }}
                    >
                        {result ? (
                            <>
                                {extractRoleTitle(result)}{" "}

                                >
                                    Run another {"\u2192"}

                        ) : (

                                    style={{
                                        width: 6,
                                        height: 6,
                                        borderRadius: "50%",
                                        background: C.green,
                                        display: "inline-block",
                                    }}

                                Free job intelligence {"\u00B7"} No account
                                required
                            </>
                        )}
                    </div>
                </div>

                {!result ? (
                    <InputState
                        jd={jd}
                        setJd={setJd}
                        companyName={companyName}
                        setCompanyName={setCompanyName}
                        jobTitle={jobTitle}
                        setJobTitle={setJobTitle}
                        loading={loading}
                        loadingMsg={loadingMsg}
                        error={error}
                        charHint={charHint()}

    )
}

function extractRoleTitle(r: AnalysisResult): string {
    const fn = r.function || ""
    const paren = fn.indexOf("(")
    return paren > 0 ? fn.slice(0, paren).trim() : fn
}
