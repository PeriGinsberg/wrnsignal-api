// ---------- Types ----------
type AnalysisResult = {
    role_level: string
    function: string
    seniority_signals: string[]
    core_skills: string[]
    hidden_requirements: string[]
    competitiveness: string
    risk_flags: string[]

type Verdict = "APPLY" | "REVIEW" | "PASS"

function getVerdict(r: AnalysisResult): Verdict {
    const comp = r.competitiveness
    const level = r.role_level
    const hiddenCount = r.hidden_requirements?.length ?? 0
    if (comp === "Very High" && hiddenCount >= 3) return "PASS"
    if ((comp === "Low" || comp === "Medium") && (level === "Entry" || level === "Early Career")) return "APPLY"
    return "REVIEW"
}

function getGenericScore(r: AnalysisResult): number {
    const compMap: Record<string, number> = { "Low": 78, "Medium": 62, "High": 45, "Very High": 30 }
    let base = compMap[r.competitiveness] ?? 50
    const hiddenCount = r.hidden_requirements?.length ?? 0
    if (hiddenCount >= 4) base -= 6
    if (r.role_level === "Mid-Level" || r.role_level === "Experienced") base -= 5
    return Math.max(15, Math.min(90, base))
}

function getCompSegments(comp: string): number {
    const map: Record<string, number> = { "Low": 1, "Medium": 2, "High": 3, "Very High": 4 }
    return map[comp] ?? 2
}

const LOADING_MSGS = [
    "Reading the job description...",
    "Identifying hidden requirements...",
    "Assessing the applicant pool...",
    "Checking what they're not telling you...",
    "Building your intelligence report...",
]

// ---------- MAIN COMPONENT ----------
export default function SignalJobAnalysis() {
    const [jd, setJd] = React.useState("")
    const [loading, setLoading] = React.useState(false)
    const [loadingMsg, setLoadingMsg] = React.useState(LOADING_MSGS[0])
    const [error, setError] = React.useState("")
    const [result, setResult] = React.useState<AnalysisResult | null>(null)

    // Track page view on mount
    React.useEffect(() => {
        if (typeof window === "undefined") return
        const existing = localStorage.getItem("jobfit_session_id")
        const sessionId = existing || crypto.randomUUID()
        if (!existing) localStorage.setItem("jobfit_session_id", sessionId)

        const url = new URL(window.location.href)
        const src = url.searchParams.get("utm_source") || sessionStorage.getItem("signal_utm_source") || null
        const med = url.searchParams.get("utm_medium") || sessionStorage.getItem("signal_utm_medium") || null
        const camp = url.searchParams.get("utm_campaign") || sessionStorage.getItem("signal_utm_campaign") || null
        try {
            if (src) sessionStorage.setItem("signal_utm_source", src)
            if (med) sessionStorage.setItem("signal_utm_medium", med)
            if (camp) sessionStorage.setItem("signal_utm_campaign", camp)
        } catch {}

        fetch(API_BASE + "/api/track", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                session_id: sessionId,
                page_path: window.location.pathname || "/analyze",
                page_name: "job_analysis_page",
                referrer: document.referrer || null,
                utm_source: src, utm_medium: med, utm_campaign: camp,
            }),
        }).catch(() => {})
    }, [])

    // Rotate loading messages
    React.useEffect(() => {
        if (!loading) return
        let idx = 0
        const iv = setInterval(() => {
            idx = (idx + 1) % LOADING_MSGS.length
            setLoadingMsg(LOADING_MSGS[idx])
        }, 2500)
        return () => clearInterval(iv)
    }, [loading])

    async function handleAnalyze() {
        setError("")
        const trimmed = jd.trim()
        if (trimmed.length < 100) {
            setError("Paste the full job description. It needs to be at least 100 characters.")
            return
        }
        setLoading(true)
        setLoadingMsg(LOADING_MSGS[0])
        try {
            const sessionId = localStorage.getItem("jobfit_session_id") || undefined
            const src = sessionStorage.getItem("signal_utm_source") || undefined
            const med = sessionStorage.getItem("signal_utm_medium") || undefined
            const camp = sessionStorage.getItem("signal_utm_campaign") || undefined

            const res = await fetch(API_BASE + "/api/job-analysis", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    job_description: trimmed,
                    session_id: sessionId,
                    utm_source: src, utm_medium: med, utm_campaign: camp,
                }),
            })
            const data = await res.json()
            if (!res.ok) throw new Error(data?.error || "Analysis failed. Try again.")
            setResult(data as AnalysisResult)
        } catch (err: any) {
            setError(err?.message || "Something went wrong. Try again.")
        } finally {
            setLoading(false)
        }
    }

    function handleCTA() {
        try {
            sessionStorage.setItem("signal_jd", JSON.stringify({
                job_description: jd.trim(),
                analysis: result,
                company_name: result?.company_name,
                timestamp: Date.now(),
            }))
        } catch {}

        // Track CTA click
        const sessionId = localStorage.getItem("jobfit_session_id") || undefined
        const src = sessionStorage.getItem("signal_utm_source") || undefined
        fetch(API_BASE + "/api/track", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ page_name: "signal_cta_click", session_id: sessionId, utm_source: src }),
        }).catch(() => {})

        window.location.assign(GHL_URL)
    }

    function handleStartOver() {
        setResult(null)
        setJd("")
        setError("")
        try { window.scrollTo({ top: 0, behavior: "smooth" }) } catch {}
    }

    function charHint(): { text: string; ready: boolean } {
        const n = jd.length
        if (n < 300) return { text: "Paste the full posting for best results", ready: false }
        if (n < 800) return { text: n + " characters -- good, paste the full posting if you have it", ready: false }
        return { text: n + " characters -- ready to analyze", ready: true }
    }

    return (
        <div style={pageStyle}>
            {/* Orbs */}
            <div style={{ ...orbBase, width: 500, height: 500, background: "rgba(81,173,229,0.1)", top: -150, right: -100 }} />
            <div style={{ ...orbBase, width: 400, height: 400, background: "rgba(254,176,106,0.07)", top: 300, left: -120 }} />
            <div style={{ ...orbBase, width: 300, height: 300, background: "rgba(192,132,252,0.07)", bottom: 100, right: -60 }} />
            {result ? <div style={{ ...orbBase, width: 280, height: 280, background: "rgba(74,222,128,0.06)", top: 900, left: -40 }} /> : null}

            <div style={{ position: "relative" as const, zIndex: 1 }}>
                {/* Topbar */}
                <div style={topbar}>
                    <div>
                        <div style={{ fontSize: 17, fontWeight: 900, letterSpacing: "-.02em" }}>
                            SIGNAL<span style={{ color: C.orange }}>--</span>
                        </div>
                        <div style={{ fontSize: 9, color: C.blue, letterSpacing: ".1em", fontWeight: 600, marginTop: 1, fontFamily: MONO }}>
                            BY WORKFORCE READY NOW
                        </div>
                    </div>
                    <div style={{ fontSize: 11, color: result ? C.muted : C.green, fontFamily: MONO, display: "flex", alignItems: "center", gap: 6 }}>
                        {result ? (
                            <>
                                {extractRoleTitle(result)} {result.company_name ? " \u00B7 " + result.company_name : ""}
                                {" \u00B7 "}
                                <span style={{ color: C.blue, cursor: "pointer" }} onClick={handleStartOver}>Run another \u2192</span>
                            </>
                        ) : (
                            <>
                                <span style={{ width: 6, height: 6, borderRadius: "50%", background: C.green, display: "inline-block" }} />
                                Free job intelligence \u00B7 No account required
                            </>
                        )}
                    </div>
                </div>

                {!result ? (
                    <InputState
                        jd={jd}
                        setJd={setJd}
                        loading={loading}
                        loadingMsg={loadingMsg}
                        error={error}
                        charHint={charHint()}
                        onAnalyze={handleAnalyze}
                    />
                ) : (
                    <ResultsState result={result} onCTA={handleCTA} onStartOver={handleStartOver} />
                )}
            </div>
        </div>
    )
}

function extractRoleTitle(r: AnalysisResult): string {
    const fn = r.function || ""
    const paren = fn.indexOf("(")
    return paren > 0 ? fn.slice(0, paren).trim() : fn
}

function WygCard({ icon, color, title, desc }: { icon: string; color: string; title: string; desc: string }) {
    const iconBg = color === C.orange ? "rgba(254,176,106,0.15)"
        : color === C.blue ? "rgba(81,173,229,0.12)"
        : color === C.purple ? "rgba(192,132,252,0.12)"
        : "rgba(74,222,128,0.1)"
    const iconBorder = color === C.orange ? "rgba(254,176,106,0.2)"
        : color === C.blue ? "rgba(81,173,229,0.2)"
        : color === C.purple ? "rgba(192,132,252,0.2)"
        : "rgba(74,222,128,0.2)"

    return (
        <div style={{ background: "rgba(255,255,255,0.03)", backdropFilter: "blur(12px)", border: "1px solid rgba(255,255,255,0.08)", borderRadius: 12, padding: "14px 16px", display: "flex", alignItems: "flex-start", gap: 12 }}>
            <div style={{ width: 32, height: 32, borderRadius: 8, flexShrink: 0, display: "flex", alignItems: "center", justifyContent: "center", background: iconBg, border: "1px solid " + iconBorder }}>
                {icon === "star" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none"><path d="M8 2L10 6H14L11 9L12 13L8 11L4 13L5 9L2 6H6L8 2Z" stroke={color} strokeWidth="1.2" strokeLinejoin="round" /></svg>
                )}
                {icon === "check" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none"><circle cx="8" cy="8" r="6" stroke={color} strokeWidth="1.2" /><path d="M5 8l2 2 4-4" stroke={color} strokeWidth="1.2" strokeLinecap="round" strokeLinejoin="round" /></svg>
                )}
                {icon === "doc" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none"><rect x="2" y="4" width="12" height="8" rx="2" stroke={color} strokeWidth="1.2" /><path d="M5 7h6M5 9.5h4" stroke={color} strokeWidth="1.2" strokeLinecap="round" /></svg>
                )}
                {icon === "tick" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none"><path d="M3 8l3.5 3.5L13 5" stroke={color} strokeWidth="1.2" strokeLinecap="round" strokeLinejoin="round" /></svg>
                )}
            </div>
            <div>
                <div style={{ fontSize: 12, fontWeight: 700, color: C.white, marginBottom: 2 }}>{title}</div>
                <div style={{ fontSize: 11, color: C.muted, lineHeight: 1.45 }}>{desc}</div>
            </div>
        </div>
    )
}
