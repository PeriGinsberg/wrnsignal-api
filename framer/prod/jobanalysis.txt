/**
 * SIGNAL Job Analysis Engine — Framer Code Component
 *
 * Two states: INPUT (paste JD) and RESULTS (full analysis).
 * Glass card aesthetic with orb glows, monospace data labels.
 * CTA links to GHL payment flow.
 */

/**
 * @framerSupportedLayoutWidth any
 * @framerSupportedLayoutHeight any
 */

// ---------- PROD CONFIG ----------
const API_BASE = "https://wrnsignal-api.vercel.app"
const GHL_URL = "https://workforcereadynow.com/signal-user-info"

// ---------- COLORS ----------
const C = {
    bg: "#0b1929",
    orange: "#FEB06A",
    blue: "#51ADE5",
    green: "#4ADE80",
    purple: "#c084fc",
    red: "#f87171",
    teal: "#2dd4bf",
    muted: "#7a9bbf",
    body: "#b0c8df",
    finding: "#ccdde8",
    white: "#fff",
    glass: "rgba(255,255,255,0.04)",
    glassBorder: "rgba(255,255,255,0.09)",
    glassBorderHover: "rgba(255,255,255,0.18)",
}

const MONO = "'Inter',system-ui,sans-serif"
const SANS = "'Inter',system-ui,sans-serif"

// ---------- Types ----------
type CompanyContext = {
    what_they_do?: string | null
    company_stage?: string | null
    clients?: string | null
    marketing_context?: string | null
    recent_news?: string | null
    application_insight?: string | null
}

type MarketReality = {
    stats?: { value: string; label: string }[]
    competitive_dynamic?: string
    // backward compat with old cached responses
    estimated_applicants?: string
    applicant_pool_description?: string
}

type AnalysisResult = {
    role_level: string
    function: string
    seniority_signals: string[]
    core_skills: string[]
    hidden_requirements: string[]
    competitiveness: string
    risk_flags: string[]
    target_candidate_profile: string[]
    summary: string
    company_name?: string
    company_context?: CompanyContext
    market_reality?: MarketReality
}

const LOADING_MSGS = [
    "Reading the job description...",
    "Identifying hidden requirements...",
    "Assessing the applicant pool...",
    "Checking what they're not telling you...",
    "Building your intelligence report...",
]

// ---------- MAIN COMPONENT ----------
export default function SignalJobAnalysis() {
    const [jd, setJd] = React.useState("")
    // Optional user-supplied company name and job title. If the JD body is
    // messy or sparse, the user can type them here and the API will honor
    // these values over any enrichment-based guess. Displayed in the
    // results header so the candidate always sees the correct labels.
    const [companyName, setCompanyName] = React.useState("")
    const [jobTitle, setJobTitle] = React.useState("")
    const [loading, setLoading] = React.useState(false)
    const [loadingMsg, setLoadingMsg] = React.useState(LOADING_MSGS[0])
    const [error, setError] = React.useState("")
    const [result, setResult] = React.useState<AnalysisResult | null>(null)
    const [capturedEmail, setCapturedEmail] = React.useState("")
    const [upgradeLoading, setUpgradeLoading] = React.useState(false)

    // Read email param from URL on mount (passed from landing page free CTA)
    React.useEffect(() => {
        try {
            const params = new URLSearchParams(window.location.search)
            const emailParam = params.get("email")
            if (emailParam) {
                setCapturedEmail(decodeURIComponent(emailParam))
            }
        } catch {
            // silently fail
        }
    }, [])

    // Track page view on mount
    React.useEffect(() => {
        if (typeof window === "undefined") return
        const existing = localStorage.getItem("jobfit_session_id")
        const sessionId = existing || crypto.randomUUID()
        if (!existing) localStorage.setItem("jobfit_session_id", sessionId)

        const url = new URL(window.location.href)
        const src =
            url.searchParams.get("utm_source") ||
            sessionStorage.getItem("signal_utm_source") ||
            null
        const med =
            url.searchParams.get("utm_medium") ||
            sessionStorage.getItem("signal_utm_medium") ||
            null
        const camp =
            url.searchParams.get("utm_campaign") ||
            sessionStorage.getItem("signal_utm_campaign") ||
            null
        const cont =
            url.searchParams.get("utm_content") ||
            sessionStorage.getItem("signal_utm_content") ||
            null
        const term =
            url.searchParams.get("utm_term") ||
            sessionStorage.getItem("signal_utm_term") ||
            null
        // Click IDs are last-touch: URL param takes precedence so a user who
        // re-lands via a different ad gets fresh attribution.
        const fbclid = url.searchParams.get("fbclid")
        const ttclid = url.searchParams.get("ttclid")
        const gclid = url.searchParams.get("gclid")
        try {
            if (src) sessionStorage.setItem("signal_utm_source", src)
            if (med) sessionStorage.setItem("signal_utm_medium", med)
            if (camp) sessionStorage.setItem("signal_utm_campaign", camp)
            if (cont) sessionStorage.setItem("signal_utm_content", cont)
            if (term) sessionStorage.setItem("signal_utm_term", term)
            if (fbclid) sessionStorage.setItem("signal_fbclid", fbclid)
            if (ttclid) sessionStorage.setItem("signal_ttclid", ttclid)
            if (gclid) sessionStorage.setItem("signal_gclid", gclid)
            // First-touch landing_page and referrer — write once so subsequent
            // intra-site navigations don't overwrite the original landing context.
            if (!sessionStorage.getItem("signal_landing_page"))
                sessionStorage.setItem(
                    "signal_landing_page",
                    window.location.pathname || ""
                )
            if (!sessionStorage.getItem("signal_referrer"))
                sessionStorage.setItem(
                    "signal_referrer",
                    document.referrer || ""
                )
        } catch {}

        fetch(API_BASE + "/api/track", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({
                session_id: sessionId,
                page_path: window.location.pathname || "/analyze",
                page_name: "job_analysis_page",
                referrer: document.referrer || null,
                utm_source: src,
                utm_medium: med,
                utm_campaign: camp,
            }),
        }).catch(() => {})
    }, [])

    // Rotate loading messages
    React.useEffect(() => {
        if (!loading) return
        let idx = 0
        const iv = setInterval(() => {
            idx = (idx + 1) % LOADING_MSGS.length
            setLoadingMsg(LOADING_MSGS[idx])
        }, 2500)
        return () => clearInterval(iv)
    }, [loading])

    async function handleAnalyze() {
        setError("")
        const trimmed = jd.trim()
        if (trimmed.length < 100) {
            setError(
                "Paste the full job description. It needs to be at least 100 characters."
            )
            return
        }
        // Company name and job title are now required — the deterministic
        // extractor has ~0% accuracy on varied JD formats so we rely on
        // the user to provide the labels we display at the top of the
        // results page.
        if (!companyName.trim()) {
            setError("Enter the company name before analyzing.")
            return
        }
        if (!jobTitle.trim()) {
            setError("Enter the job title before analyzing.")
            return
        }
        setLoading(true)
        setLoadingMsg(LOADING_MSGS[0])
        try {
            const sessionId =
                localStorage.getItem("jobfit_session_id") || undefined
            const src = sessionStorage.getItem("signal_utm_source") || undefined
            const med = sessionStorage.getItem("signal_utm_medium") || undefined
            const camp =
                sessionStorage.getItem("signal_utm_campaign") || undefined

            const res = await fetch(API_BASE + "/api/job-analysis", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    job_description: trimmed,
                    // User-provided labels are required — validation above
                    // ensures both are non-empty before we hit this point.
                    company_name: companyName.trim(),
                    job_title: jobTitle.trim(),
                    session_id: sessionId,
                    utm_source: src,
                    utm_medium: med,
                    utm_campaign: camp,
                }),
            })
            const data = await res.json()
            if (!res.ok)
                throw new Error(data?.error || "Analysis failed. Try again.")
            setResult(data as AnalysisResult)
        } catch (err: any) {
            setError(err?.message || "Something went wrong. Try again.")
        } finally {
            setLoading(false)
        }
    }

    function handleCTA() {
        try {
            sessionStorage.setItem(
                "signal_jd",
                JSON.stringify({
                    job_description: jd.trim(),
                    analysis: result,
                    company_name: result?.company_name,
                    timestamp: Date.now(),
                })
            )
        } catch {}

        // Track upgrade CTA click (analyzer results -> GHL purchase).
        // Uses sendBeacon so the event survives the redirect. We send as
        // text/plain (not application/json) because application/json triggers
        // a CORS preflight that sendBeacon cannot wait for during page unload,
        // so Chrome silently drops the beacon. The server parses the body via
        // req.json() which ignores Content-Type, so this is safe.
        const sessionId = localStorage.getItem("jobfit_session_id") || undefined
        const src = sessionStorage.getItem("signal_utm_source") || undefined
        const payload = JSON.stringify({
            page_name: "upgrade_cta_click",
            session_id: sessionId,
            page_path: window.location.pathname,
            utm_source: src,
        })
        try {
            if (typeof navigator !== "undefined" && navigator.sendBeacon) {
                const blob = new Blob([payload], { type: "text/plain" })
                navigator.sendBeacon(API_BASE + "/api/track", blob)
            } else {
                fetch(API_BASE + "/api/track", {
                    method: "POST",
                    headers: { "Content-Type": "text/plain" },
                    body: payload,
                    keepalive: true,
                }).catch(() => {})
            }
        } catch {}

        // Small delay lets the beacon queue before navigation. sendBeacon is
        // supposed to survive unload, but in practice ~50ms makes delivery
        // much more reliable across browsers.
        setTimeout(() => {
            window.top!.location.replace("https://wrnsignal.workforcereadynow.com/signal/jobfit")
        }, 50)
    }

    function handleStartOver() {
        setResult(null)
        setJd("")
        setCompanyName("")
        setJobTitle("")
        setError("")
        try {
            window.scrollTo({ top: 0, behavior: "smooth" })
        } catch {}
    }

    // sessionStorage read with try/catch — private mode + SSR safe. "" on miss.
    function ss(key: string): string {
        try {
            return sessionStorage.getItem(key) || ""
        } catch {
            return ""
        }
    }

    // Cookie read by name. "" if missing. _fbp/_fbc/_ttp will be "" until
    // Meta/TikTok pixels are installed; webhook synthesizes fbc from fbclid.
    function getCookie(name: string): string {
        if (typeof document === "undefined") return ""
        const m = document.cookie.match(
            new RegExp("(?:^|; )" + name.replace(/[-.]/g, "\\$&") + "=([^;]*)")
        )
        return m ? decodeURIComponent(m[1]) : ""
    }

    // All fields default to "" so the POST body is well-formed and the
    // server can treat missing == empty uniformly.
    function getAttributionSnapshot() {
        return {
            utm_source: ss("signal_utm_source"),
            utm_medium: ss("signal_utm_medium"),
            utm_campaign: ss("signal_utm_campaign"),
            utm_content: ss("signal_utm_content"),
            utm_term: ss("signal_utm_term"),
            landing_page: ss("signal_landing_page"),
            referrer: ss("signal_referrer"),
            fbclid: ss("signal_fbclid"),
            ttclid: ss("signal_ttclid"),
            gclid: ss("signal_gclid"),
            fbp: getCookie("_fbp"),
            fbc: getCookie("_fbc"),
            ttp: getCookie("_ttp"),
        }
    }

    async function handleUpgrade() {
        if (upgradeLoading) return
        if (!capturedEmail) {
            window.top!.location.replace("https://wrnsignal.workforcereadynow.com/signal/jobfit")
            return
        }
        setUpgradeLoading(true)
        try {
            const response = await fetch(
                "https://wrnsignal-api.vercel.app/api/checkout/create-session",
                {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({
                        email: capturedEmail,
                        ...getAttributionSnapshot(),
                    }),
                }
            )
            const data = await response.json()
            if (data.url) {
                window.open(data.url, "_blank")
            } else {
                window.top!.location.replace("https://wrnsignal.workforcereadynow.com/signal/jobfit")
            }
        } catch {
            window.top!.location.replace("https://wrnsignal.workforcereadynow.com/signal/jobfit")
        } finally {
            setUpgradeLoading(false)
        }
    }

    function charHint(): { text: string; ready: boolean } {
        const n = jd.length
        if (n < 300)
            return {
                text: "Paste the full posting for best results",
                ready: false,
            }
        if (n < 800)
            return {
                text:
                    n +
                    " characters \u2014 good, paste the full posting if you have it",
                ready: false,
            }
        return { text: n + " characters \u2014 ready to analyze", ready: true }
    }

    return (
        <div style={pageStyle}>
            {/* Injected keyframes */}
            <style
                dangerouslySetInnerHTML={{
                    __html: `
                @keyframes ctaGlow {
                    0%, 100% { box-shadow: 0 0 20px rgba(254,176,106,0.4), 0 0 40px rgba(254,176,106,0.15); }
                    50% { box-shadow: 0 0 28px rgba(81,173,229,0.5), 0 0 56px rgba(81,173,229,0.2); }
                }
                .signal-cta-btn {
                    animation: ctaGlow 3s ease-in-out infinite;
                    transition: transform 0.2s ease, box-shadow 0.2s ease;
                }
                .signal-cta-btn:hover {
                    transform: scale(1.03);
                    box-shadow: 0 0 32px rgba(254,176,106,0.6), 0 0 64px rgba(81,173,229,0.3) !important;
                }
            `,
                }}
            />
            {/* Orbs */}
            <div
                style={{
                    ...orbBase,
                    width: 500,
                    height: 500,
                    background: "rgba(81,173,229,0.1)",
                    top: -150,
                    right: -100,
                }}
            />
            <div
                style={{
                    ...orbBase,
                    width: 400,
                    height: 400,
                    background: "rgba(254,176,106,0.07)",
                    top: 300,
                    left: -120,
                }}
            />
            <div
                style={{
                    ...orbBase,
                    width: 300,
                    height: 300,
                    background: "rgba(192,132,252,0.07)",
                    bottom: 100,
                    right: -60,
                }}
            />
            {result ? (
                <div
                    style={{
                        ...orbBase,
                        width: 280,
                        height: 280,
                        background: "rgba(74,222,128,0.06)",
                        top: 900,
                        left: -40,
                    }}
                />
            ) : null}

            <div style={{ position: "relative" as const, zIndex: 1 }}>
                {/* Topbar */}
                <div style={topbar}>
                    <div>
                        <div
                            style={{
                                fontSize: 17,
                                fontWeight: 900,
                                letterSpacing: "-.02em",
                            }}
                        >
                            SIGNAL
                            <span style={{ color: C.orange }}>{"\u2014"}</span>
                        </div>
                        <div
                            style={{
                                fontSize: 9,
                                color: C.blue,
                                letterSpacing: ".1em",
                                fontWeight: 600,
                                marginTop: 1,
                                fontFamily: MONO,
                            }}
                        >
                            BY WORKFORCE READY NOW
                        </div>
                    </div>
                    <div
                        style={{
                            fontSize: 11,
                            color: result ? "#4a6a85" : C.green,
                            display: "flex",
                            alignItems: "center",
                            gap: 6,
                        }}
                    >
                        {result ? (
                            <>
                                {extractRoleTitle(result)}{" "}
                                {result.company_name
                                    ? " \u00B7 " + result.company_name
                                    : ""}
                                {" \u00B7 "}
                                <span
                                    style={{ color: C.blue, cursor: "pointer" }}
                                    onClick={handleStartOver}
                                >
                                    Run another {"\u2192"}
                                </span>
                            </>
                        ) : (
                            <>
                                <span
                                    style={{
                                        width: 6,
                                        height: 6,
                                        borderRadius: "50%",
                                        background: C.green,
                                        display: "inline-block",
                                    }}
                                />
                                Free job intelligence {"\u00B7"} No account
                                required
                            </>
                        )}
                    </div>
                </div>

                {!result ? (
                    <InputState
                        jd={jd}
                        setJd={setJd}
                        companyName={companyName}
                        setCompanyName={setCompanyName}
                        jobTitle={jobTitle}
                        setJobTitle={setJobTitle}
                        loading={loading}
                        loadingMsg={loadingMsg}
                        error={error}
                        charHint={charHint()}
                        onAnalyze={handleAnalyze}
                        onUpgrade={handleUpgrade}
                        upgradeLoading={upgradeLoading}
                    />
                ) : (
                    <ResultsState
                        result={result}
                        jobTitle={jobTitle}
                        companyName={companyName}
                        onStartOver={handleStartOver}
                        onUpgrade={handleUpgrade}
                        upgradeLoading={upgradeLoading}
                    />
                )}
            </div>
        </div>
    )
}

function extractRoleTitle(r: AnalysisResult): string {
    const fn = r.function || ""
    const paren = fn.indexOf("(")
    return paren > 0 ? fn.slice(0, paren).trim() : fn
}

function WygCard({
    icon,
    color,
    title,
    desc,
}: {
    icon: string
    color: string
    title: string
    desc: string
}) {
    const iconBg =
        color === C.orange
            ? "rgba(254,176,106,0.15)"
            : color === C.blue
              ? "rgba(81,173,229,0.12)"
              : color === C.purple
                ? "rgba(192,132,252,0.12)"
                : "rgba(74,222,128,0.1)"
    const iconBorder =
        color === C.orange
            ? "rgba(254,176,106,0.2)"
            : color === C.blue
              ? "rgba(81,173,229,0.2)"
              : color === C.purple
                ? "rgba(192,132,252,0.2)"
                : "rgba(74,222,128,0.2)"

    return (
        <div
            style={{
                background: "rgba(255,255,255,0.03)",
                backdropFilter: "blur(12px)",
                border: "1px solid rgba(255,255,255,0.08)",
                borderRadius: 12,
                padding: "14px 16px",
                display: "flex",
                alignItems: "flex-start",
                gap: 12,
            }}
        >
            <div
                style={{
                    width: 32,
                    height: 32,
                    borderRadius: 8,
                    flexShrink: 0,
                    display: "flex",
                    alignItems: "center",
                    justifyContent: "center",
                    background: iconBg,
                    border: "1px solid " + iconBorder,
                }}
            >
                {icon === "star" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none">
                        <path
                            d="M8 2L10 6H14L11 9L12 13L8 11L4 13L5 9L2 6H6L8 2Z"
                            stroke={color}
                            strokeWidth="1.2"
                            strokeLinejoin="round"
                        />
                    </svg>
                )}
                {icon === "check" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none">
                        <circle
                            cx="8"
                            cy="8"
                            r="6"
                            stroke={color}
                            strokeWidth="1.2"
                        />
                        <path
                            d="M5 8l2 2 4-4"
                            stroke={color}
                            strokeWidth="1.2"
                            strokeLinecap="round"
                            strokeLinejoin="round"
                        />
                    </svg>
                )}
                {icon === "doc" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none">
                        <rect
                            x="2"
                            y="4"
                            width="12"
                            height="8"
                            rx="2"
                            stroke={color}
                            strokeWidth="1.2"
                        />
                        <path
                            d="M5 7h6M5 9.5h4"
                            stroke={color}
                            strokeWidth="1.2"
                            strokeLinecap="round"
                        />
                    </svg>
                )}
                {icon === "tick" && (
                    <svg width="16" height="16" viewBox="0 0 16 16" fill="none">
                        <path
                            d="M3 8l3.5 3.5L13 5"
                            stroke={color}
                            strokeWidth="1.2"
                            strokeLinecap="round"
                            strokeLinejoin="round"
                        />
                    </svg>
                )}
            </div>
            <div>
                <div
                    style={{
                        fontSize: 12,
                        fontWeight: 700,
                        color: C.white,
                        marginBottom: 2,
                    }}
                >
                    {title}
                </div>
                <div style={{ fontSize: 11, color: C.muted, lineHeight: 1.45 }}>
                    {desc}
                </div>
            </div>
        </div>
    )
}
