        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ email }),
      })
      const data = await response.json()
      if (data.url) {
        setState("success")
        setTimeout(() => { window.top!.location.href = data.url }, 500)
      } else {
        setError("Something went wrong. Please try again.")
        setState("error")
      }
    } catch {
      setError("Something went wrong. Please try again.")
      setState("error")
    }
