function safePreview(s: string, max = 1400) {
    const t = (s || "").trim()
    return t.length > max ? t.slice(0, max) + "\n…(truncated)" : t
}

function looksLikeHtml(text: string) {
    const t = (text || "").trim().toLowerCase()
    return (
        t.startsWith("<!doctype") ||
        t.startsWith("<html") ||
        t.includes("<head") ||
        t.includes("<body")
    )
}

async function postJsonWithDebug(
    url: string,
    payload: any,
    onDebug: (d: any) => void,
    accessToken?: string | null
) {
    const startedAt = Date.now()
    let res: Response

    const headers: Record<string, string> = {
        "Content-Type": "application/json",
    }
    if (accessToken) headers["Authorization"] = `Bearer ${accessToken}`

    try {
        res = await fetch(url, {
            method: "POST",
            headers,
            body: JSON.stringify(payload),
        })
    } catch (err: any) {
        onDebug({
            stage: "fetch_throw",
            url,
            ms: Date.now() - startedAt,
            errorName: err?.name,
            errorMessage: err?.message,
            errorStackPreview: safePreview(err?.stack || "", 900),
        })
        throw new Error(`Request failed. Please try again.`)
    }

    const ms = Date.now() - startedAt
    const contentType = res.headers.get("content-type") || ""
    const raw = await res.text()

    onDebug({
        stage: "fetch_response",
        url,
        ms,
        status: res.status,
        ok: res.ok,
        contentType,
        responsePreview: safePreview(raw),
    })

    if (looksLikeHtml(raw) || contentType.includes("text/html")) {
        throw new Error(`Unexpected response. Please try again.`)
    }

    let data: any = null
    try {
        data = JSON.parse(raw)
    } catch {
        throw new Error(`Unexpected response. Please try again.`)
    }

    if (!res.ok) {
        throw new Error(
            data?.detail || data?.error || `Request failed. Please try again.`
        )
    }

    return data
}

/* ---------- DIFF HIGHLIGHT (TEAL) ---------- */
function renderDiffTeal(before: string, after: string) {
    const normalize = (w: string) =>
        w
            .toLowerCase()
            .replace(/^[^a-z0-9]+|[^a-z0-9]+$/gi, "")
            .trim()

    const stop = new Set([
        "a",
        "an",
        "the",
        "and",
        "or",
        "but",
        "to",
        "of",
        "in",
        "on",
        "for",
        "with",
        "at",
        "by",
        "from",
        "as",
        "is",
        "are",
        "was",
        "were",
        "be",
        "been",
        "being",
        "this",
        "that",
        "these",
        "those",
        "it",
        "its",
        "you",
        "your",
        "we",
        "our",
        "i",
    ])

    const beforeWords = new Set(
        (before || "")
            .split(/\s+/)
            .map(normalize)
            .filter((w) => w && w.length >= 4 && !stop.has(w))
    )

    const parts = (after || "").split(/(\b[\w'/-]+\b)/g)

    return (
        <>
            {parts.map((p, i) => {
                if (!p || !/\b[\w'/-]+\b/.test(p))
                    return <span key={i}>{p}</span>

                const w = normalize(p)
                const shouldBold =
                    !!w && w.length >= 4 && !stop.has(w) && !beforeWords.has(w)

                return shouldBold ? (
                    <span key={i} style={{ fontWeight: 950, color: WRN_TEAL }}>
                        {p}
                    </span>
                ) : (
                    <span key={i}>{p}</span>
                )
            })}
        </>
    )
}

function extractJobContextFromJobText(jobText: string) {
    const text = String(jobText || "")
        .replace(/\r/g, "")
        .trim()
    if (!text) return { title: "", company: "" }

    const lines = text
        .split("\n")
        .map((l) => l.trim())
        .filter(Boolean)

    const firstN = lines.slice(0, 40)

    const clean = (s: string) =>
        String(s || "")
            .replace(/[–—]/g, "-")
            .replace(/\s+/g, " ")
            .trim()

    const looksLikeMarketingFluff = (s: string) => {
        const t = (s || "").toLowerCase()
        const phrases = [
            "best place to work",
            "great place to work",
            "fastest growing",
            "award",
            "top workplace",
            "fortune",
            "inc.",
            "recognition",
            "ranked",
            "named",
        ]
        if (phrases.some((p) => t.includes(p))) return true
        const quoteCount = (s.match(/["“”']/g) || []).length
        if (quoteCount >= 4) return true
        const commaCount = (s.match(/,/g) || []).length
        const words = s.split(/\s+/).filter(Boolean)
        const upperWords = words.filter((w) => /^[A-Z]{3,}$/.test(w)).length
        if (commaCount >= 2 && upperWords >= 3) return true
        return false
    }

    const looksLikeRealRoleTitle = (s: string) => {
        const t = (s || "").toLowerCase()
        const roleWords = [
            "intern",
            "analyst",
            "associate",
            "manager",
            "director",
            "coordinator",
            "specialist",
            "engineer",
            "consultant",
            "developer",
            "designer",
            "strategist",
            "assistant",
            "representative",
            "officer",
            "lead",
            "head",
        ]
        return roleWords.some((w) => t.includes(w))
    }

    const isJunkTitle = (s: string) => {
        const t = (s || "").toLowerCase()
        if (!t) return true
        if (t.length < 4) return true
        if (t.length > 90) return true
        if (looksLikeMarketingFluff(s)) return true
        if (
            t.includes("overview") ||
            t.includes("responsibilities") ||
            t.includes("qualifications") ||
            t.includes("requirements") ||
            t.includes("equal opportunity") ||
            t.includes("accommodation")
        )
            return true
        if (!looksLikeRealRoleTitle(s)) return true
        return false
    }

    const isJunkCompany = (s: string) => {
        const t = s.toLowerCase()
        if (!t) return true
        if (t.length < 2) return true
        if (t.length > 70) return true
        if (
            t.includes("overview") ||
            t.includes("position overview") ||
            t.includes("responsibilities") ||
            t.includes("qualifications") ||
            t.includes("equal opportunity")
        )
            return true
        return false
    }

    let company = ""

    for (const l of firstN.slice(0, 20)) {
        const m = l.match(/^([A-Z][A-Za-z0-9&.'-]{2,40})\s+Overview$/)
        if (m?.[1]) {
            const c = clean(m[1])
            if (!isJunkCompany(c)) company = company || c
        }
    }

    if (!company) {
        for (const l of firstN) {
            const m = l.match(
                /^([A-Z][A-Za-z0-9&.'-]{2,60})\s+is\s+(?:currently\s+)?looking\b/i
            )
            if (m?.[1]) {
                const c = clean(m[1])
                if (!isJunkCompany(c)) {
                    company = c
                    break
                }
            }
        }
    }

    if (!company) {
        for (const l of firstN) {
            const m = l.match(/^(Company|Employer)\s*:\s*(.+)$/i)
            if (m?.[2]) {
                const c = clean(m[2])
                if (!isJunkCompany(c)) {
                    company = c
                    break
                }
            }
        }
    }

    let title = ""

    for (const l of firstN) {
        const m = l.match(/^(Position Title|Job Title|Role)\s*:\s*(.+)$/i)
        if (m?.[2]) {
            const t = clean(m[2])
            if (!isJunkTitle(t)) {
                title = t
                break
            }
        }
    }

    if (!title) {
        for (const l of firstN) {
            const m = l.match(/\bas\s+(?:an?|the)\s+(.{4,80})$/i)
            if (m?.[1]) {
                const t = clean(m[1])
                if (!isJunkTitle(t)) {
                    title = t
                    break
                }
            }
        }
    }

    title = clean(title)
    company = clean(company)

    if (isJunkTitle(title)) title = ""
    if (isJunkCompany(company)) company = ""

    return { title, company }
}

/* ---------- TEXT HELPERS (SAFE) ---------- */
function pickString(obj: any, keys: string[]) {
    for (const k of keys) {
        const v = obj?.[k]
        if (typeof v === "string" && v.trim()) return v.trim()
    }
    return ""
}

function toSafeLine(x: any) {
    if (x === null || x === undefined) return ""
    if (typeof x === "string") return x.trim()
    if (typeof x === "number") return String(x)
    try {
        return JSON.stringify(x)
    } catch {
        return String(x)
    }
}

function pickFirstArray(...vals: any[]) {
    for (const v of vals) {
        if (Array.isArray(v) && v.length) return v
    }
    return []
}

/* ---------- COVER LETTER HELPERS ---------- */
function normalizeLetterBody(raw: string) {
    const s = String(raw || "")
        .replace(/\r/g, "")
        .trim()
    if (!s) return ""

    const lines = s.split("\n").map((l) => l.replace(/[ \t]+$/g, ""))
    const out: string[] = []
    let lastWasBlank = false

    for (const line of lines) {
        const isBlank = !line.trim()
        if (isBlank) {
            if (!lastWasBlank) out.push("")
            lastWasBlank = true
        } else {
            out.push(line.trimStart())
            lastWasBlank = false
        }
    }

    while (out.length && !out[0].trim()) out.shift()
    while (out.length && !out[out.length - 1].trim()) out.pop()

    return out.join("\n")
}

// Strip any leading salutation like "Dear X," so we never print two "Dear" lines.
function stripLeadingSalutation(raw: string) {
    const s = normalizeLetterBody(raw)
    if (!s) return ""

    const lines = s.split("\n")
    const firstNonBlankIndex = lines.findIndex((l) => l.trim().length > 0)
    if (firstNonBlankIndex === -1) return s

    const first = lines[firstNonBlankIndex].trim()
    const isDear = /^dear\s+.+/i.test(first)

    if (!isDear) return s

    // Remove that line and any immediate blank line after it
    const remaining = lines.slice(firstNonBlankIndex + 1)
    while (remaining.length && !remaining[0].trim()) remaining.shift()

    return normalizeLetterBody(remaining.join("\n"))
}

function stripExistingSignatureBlock(raw: string) {
    const s = normalizeLetterBody(raw)
    if (!s) return ""

    const signoffs = [
        "sincerely,",
        "sincerely",
        "best,",
        "best regards,",
        "best regards",
        "regards,",
        "regards",
        "thank you,",
        "thank you",
        "respectfully,",
        "respectfully",
    ]

    const lines = s.split("\n")
    const start = Math.max(0, lines.length - 10)

    for (let i = lines.length - 1; i >= start; i--) {
        const t = lines[i].trim().toLowerCase()
        if (signoffs.includes(t)) {
            return normalizeLetterBody(lines.slice(0, i).join("\n"))
        }
        for (const so of signoffs) {
            if (t.startsWith(so)) {
                return normalizeLetterBody(lines.slice(0, i).join("\n"))
            }
        }
    }

    return s
}

function buildFinalLetter(args: {
    bodyRaw: string
    jobTitle: string
    companyName: string
    fullName: string
    phone: string
    email: string
}) {
    const today = new Date().toLocaleDateString("en-US", {
        year: "numeric",
        month: "long",
        day: "numeric",
    })

    // Fix: remove salutation + signature inside the model body
    const bodyNoDear = stripLeadingSalutation(args.bodyRaw)
    const body = stripExistingSignatureBlock(bodyNoDear)

    // If you are temporarily not using company/title, blank these out later.
    const hiringManager = "[Hiring Manager]"
    const company = args.companyName || "[Company Name]"
    const title = args.jobTitle || "[Position Title]"

    // IMPORTANT: only use args here. Never reference React state variables inside this helper.
    const name = String(args.fullName || "").trim() || "[Your Name]"
    const phone = String(args.phone || "").trim() || "[Phone Number]"
    const email = String(args.email || "").trim() || "[Email Address]"

    // If your model body still includes an opener like "I am applying for X at Y", fine.
    // This template only guarantees the header + date + greeting + cleaned body + signature.

    return `${name}
${phone}
${email}

${today}

${hiringManager}
${company}

Dear ${hiringManager},

${body}

Sincerely,
${name}
`
}

/* ---------- NETWORKING EXTRACTORS ---------- */
function extractNetworkingTarget(a: any) {
    if (!a) return ""

    const roles = a?.target_roles
    if (Array.isArray(roles) && roles.length) {
        const first =
            roles.find((x: any) => typeof x === "string" && x.trim()) ||
            roles.find(
                (x: any) => typeof x?.title === "string" && x.title.trim()
            )
        if (typeof first === "string") return first.trim()
        if (first?.title) return String(first.title).trim()
    } else if (typeof roles === "string" && roles.trim()) {
        return roles.trim()
    }

    const person = pickString(a, ["person_to_pick", "personToPick"])
    if (person) return person

    const terms = a?.search_terms
    if (Array.isArray(terms) && terms.length) {
        const firstTerm = terms.find(
            (x: any) => typeof x === "string" && x.trim()
        )
        if (firstTerm) return firstTerm.trim()
    } else if (typeof terms === "string" && terms.trim()) {
        return terms.trim()
    }

    return pickString(a, [
        "linkedin_title",
        "linkedinTitle",
        "search_title",
        "searchTitle",
        "target_title",
        "targetTitle",
        "role",
        "title",
        "persona",
        "label",
    ])
}

function extractNetworkingMessageParts(action: any) {
    const msg =
        action?.message ??
        action?.outreach_message ??
        action?.outreachMessage ??
        action?.copy ??
        action?.draft ??
        ""

    if (typeof msg === "string") {
        return { initial: msg.trim(), followUp: "" }
    }

    if (msg && typeof msg === "object") {
        const initial =
            pickString(msg, ["initial", "first", "message", "text", "body"]) ||
            pickString(msg?.content, ["text", "body", "content"]) ||
            pickString(msg?.message, ["text", "body", "content"]) ||
            ""

        const followUp =
            pickString(msg, ["follow_up", "followUp", "followup", "second"]) ||
            pickString(msg?.follow_up, ["text", "body", "content"]) ||
            pickString(msg?.followUp, ["text", "body", "content"]) ||
            ""

        return { initial, followUp }
    }

    return { initial: "", followUp: "" }
}

                    const code = url.searchParams.get("code")

                    if (code) {
                        const { error } =
                            await supabase.auth.exchangeCodeForSession(code)

                        url.searchParams.delete("code")
                        window.history.replaceState({}, "", url.toString())

                        if (error) {
                            console.warn(
                                "[auth] exchangeCodeForSession failed:",
                                error.message
                            )
                        }
                    }
                }

                const { data, error } = await supabase.auth.getSession()
                if (!mounted) return

                if (error) {
                    setAccessToken(null)
                    setAuthedEmail(null)
                    setAuthStatus("logged_out")
                    setAccessMode(computeAccessMode(null))
                    return
                }

                const session = data?.session
                if (session?.access_token) {
                    setAccessToken(session.access_token)
                    setAuthedEmail(session.user?.email || null)
                    setAuthStatus("logged_in")
                    setAccessMode(computeAccessMode(session.access_token))
                } else {
                    setAccessToken(null)
                    setAuthedEmail(null)
                    setAuthStatus("logged_out")
                    setAccessMode(computeAccessMode(null))
                }
            } catch {
                if (!mounted) return
                setAccessToken(null)
                setAuthedEmail(null)
                setAuthStatus("logged_out")
                setAccessMode(computeAccessMode(null))
            }
        }

        initSession()

        const { data: sub } = supabase.auth.onAuthStateChange(
            (_event, session) => {
                if (!mounted) return
                if (session?.access_token) {
                    setAccessToken(session.access_token)
                    setAuthedEmail(session.user?.email || null)
                    setAuthStatus("logged_in")
                    setAccessMode("full")
                } else {
                    setAccessToken(null)
                    setAuthedEmail(null)
                    setAuthStatus("logged_out")
                    setAccessMode(computeAccessMode(null))
                }
            }
        )

        function onTrialUnlock() {
            if (!mounted) return
            if (!accessTokenRef.current) setAccessMode("jobfit_only")
        }

        if (typeof window !== "undefined") {
            window.addEventListener("jobfit_unlocked", onTrialUnlock)
        }

        return () => {
            mounted = false
            sub?.subscription?.unsubscribe?.()
            if (typeof window !== "undefined") {
                window.removeEventListener("jobfit_unlocked", onTrialUnlock)
            }
        }
    }, [])

        if (authStatus !== "checking") return
        const t = setTimeout(() => {
            setAuthStatus("logged_out")
            setAccessMode(computeAccessMode(null))
        }, 2500)
        return () => clearTimeout(t)
    }, [authStatus])

    async function logout() {
        // 1️⃣ Clear any error/debug state
        clearErrorAndDebug()

        try {
            // 2️⃣ Sign out from Supabase (full-access users)
            if (supabase) await supabase.auth.signOut()
        } catch (err) {
            console.warn("Supabase signOut error:", err)
        } finally {
            // 3️⃣ Reset React state
            setAccessToken(null)
            setAuthedEmail(null)
            setAuthStatus("logged_out")
            setAccessMode("full") // default for new session
            runNewJob(true)

            // 4️⃣ Clear trial info from localStorage
            if (typeof window !== "undefined") {
                localStorage.removeItem("jobfit_email")
                localStorage.removeItem("jobfit_unlocked")
            }

            // 5️⃣ Force UI refresh to clear SideNav
            if (typeof window !== "undefined") window.location.reload()
        }
    }

    async function runJobFit() {
        clearErrorAndDebug()
        setLoadingJobFit(true)

        // Pre-capture context from job text (fallback) but do not overwrite existing
        setJobContext((prev) => {
            if (prev.title || prev.company) return prev
            const fromText = extractJobContextFromJobText(job)
            if (fromText.title || fromText.company) return fromText
            return prev
        })

        try {
            if (!hasJob) throw new Error("Paste a job description first.")

            if (accessMode === "jobfit_only") {
                const trialEmail = getTrialEmail().trim().toLowerCase()
                if (!trialEmail) {
                    throw new Error(
                        "Missing email. Complete the Job Fit intake first so we can save your profile."
                    )
                }

                const data = await postJsonWithDebug(
                    `${API_BASE}/api/jobfit-run-trial`,
                    { email: trialEmail, job_description: job.trim() },
                    setDebug,
                    null
                )

                const normalized = normalizeJobFitResponseForUI(data)
                setJobFitResult(normalized)

                const ctxFromApi = extractJobContextFromJobFit(normalized)
                const ctxFromText = extractJobContextFromJobText(job)

                setJobContext(() => {
                    if (ctxFromApi.title || ctxFromApi.company)
                        return ctxFromApi
                    if (ctxFromText.title || ctxFromText.company)
                        return ctxFromText
                    return { title: "", company: "" }
                })

                setTab("jobfit")
                return
            }

            if (!accessToken) throw new Error("Please log in again.")

            const data = await postJsonWithDebug(
                `${API_BASE}/api/jobfit`,
                { job: job.trim() },
                setDebug,
                accessToken
            )

            const normalized = normalizeJobFitResponseForUI(data)
            setJobFitResult(normalized)

            const ctxFromApi = extractJobContextFromJobFit(normalized)
            setJobContext((prev) => {
                if (ctxFromApi.title || ctxFromApi.company) return ctxFromApi
                return prev
            })

            setTab("jobfit")
        } catch (e: any) {
            setErrorMsg(e?.message || String(e))
        } finally {
            setLoadingJobFit(false)
        }
    }

    const renderMainOutput = () => {

                                    <div style={stepRow}>
                                        <div style={stepNum}>2</div>

/* ---------- SIDE NAV ---------- */
function SideNav({
    tab,
    accessMode,
    isLoggedIn,
    authedEmail,
    trialEmail,
    onSelect,
    onRunNewJob,
    onLogout,
    onUpgrade,
}: {
    tab: TabKey
    accessMode: AccessMode
    isLoggedIn: boolean
    authedEmail: string | null
    trialEmail: string
    onSelect: (t: TabKey) => void
    onRunNewJob: () => void
    onLogout: () => void
    onUpgrade: () => void
}) {
    const locked = (k: TabKey) => accessMode === "jobfit_only" && k !== "jobfit"

    return (
        <div style={sideNav}>
            <div style={sideNavTopSpacer} />

            <div style={sideSectionTitle}>SYSTEM</div>

            <SideNavItem
                label="JobFit"
                active={tab === "jobfit"}
                locked={false}
                onClick={() => onSelect("jobfit")}
            />
            <SideNavItem
                label="Positioning"
                active={tab === "positioning"}
                locked={locked("positioning")}
                onClick={() => onSelect("positioning")}
            />
            <SideNavItem
                label="Cover Letter"
                active={tab === "coverletter"}
                locked={locked("coverletter")}
                onClick={() => onSelect("coverletter")}
            />
            <SideNavItem
                label="Networking"
                active={tab === "networking"}
                locked={locked("networking")}
                onClick={() => onSelect("networking")}
            />

            <div style={{ height: 14 }} />

            <div style={sideSectionTitle}>ACTIONS</div>

            <button style={sideActionBtn} onClick={onRunNewJob} type="button">
                Run New Job
            </button>

            {accessMode === "jobfit_only" ? (
                <button
                    style={sideActionBtnSecondary}
                    onClick={onUpgrade}
                    type="button"
                >
                    Unlock Full SIGNAL
                </button>
            ) : null}

            <div style={{ height: 14 }} />

            <div style={sideSectionTitle}>ACCOUNT</div>

            <div style={sideAccountRow}>
                <div style={sideAvatar}>
                    {(authedEmail || trialEmail || "?").charAt(0).toUpperCase()}
                </div>
                <div style={{ overflow: "hidden" }}>
                    <div style={sideAccountName}>
                        {isLoggedIn
                            ? "Signed in"
                            : trialEmail
                              ? "Job Fit trial"
                              : "Guest"}
                    </div>
                    <div style={sideAccountEmail}>
                        {authedEmail || trialEmail || "Trial mode"}
                    </div>
                </div>
            </div>
            <button style={sideLogoutBtn} onClick={onLogout} type="button">
                Log Out
            </button>

            <div style={{ height: 18 }} />

            <div style={sideSectionTitle}>PLACEHOLDERS</div>
            <div style={sideDisabledItem}>Profile (future)</div>
            <div style={sideDisabledItem}>Saved Jobs (future)</div>
        </div>
    )
}

function SideNavItem({
    label,
    active,
    locked,
    onClick,
}: {
    label: string
    active: boolean
    locked: boolean
    onClick: () => void
}) {
    return (
        <button
            style={{
                ...sideNavItem,
                ...(active ? sideNavItemActive : null),
                opacity: locked ? 0.35 : 1,
                cursor: locked ? "not-allowed" : "pointer",
            }}
            onClick={locked ? undefined : onClick}
            disabled={locked}
            title={locked ? "Locked" : undefined}
            type="button"
        >
            <span>{label}</span>
            <span style={{ opacity: 0.8 }}>{locked ? "🔒" : "›"}</span>
        </button>
    )
}
