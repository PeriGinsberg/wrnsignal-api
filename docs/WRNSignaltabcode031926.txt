// ---------- Types ----------
type Step = 1 | 2 | 3

function sleep(ms: number) {
    return new Promise((r) => setTimeout(r, ms))
}

function isBrowser() {
    return typeof window !== "undefined"
}

function hasAuthStuffInUrl() {
    if (!isBrowser()) return false
    const href = window.location.href
    if (href.includes("?code=")) return true
    const h = window.location.hash || ""
    return (
        h.includes("access_token=") ||
        h.includes("refresh_token=") ||
        h.includes("type=magiclink")
    )
}

function cleanAuthFromUrl() {
    if (!isBrowser()) return
    try {
        const url = new URL(window.location.href)
        if (url.searchParams.has("code")) url.searchParams.delete("code")
        const cleaned = url.pathname + (url.search ? url.search : "")
        window.history.replaceState({}, document.title, cleaned)
    } catch {}
}

async function tryExchangeCodeForSessionIfPresent() {
    if (!isBrowser()) return
    try {
        const url = new URL(window.location.href)
        const code = url.searchParams.get("code")
        if (!code) return
        await supabase.auth.exchangeCodeForSession(code)
    } catch {}
}

async function getSessionWithRetry() {
    const attempts = [0, 75, 150, 300]
    let last: any = null

    for (let i = 0; i < attempts.length; i++) {
        if (attempts[i] > 0) await sleep(attempts[i])
        const res = await supabase.auth.getSession()
        last = res
        const token = res?.data?.session?.access_token
        if (token) return res
    }

    return last
}

export default function SignalIntakeFullAccess() {
    const [booting, setBooting] = React.useState(true)
    const [authed, setAuthed] = React.useState(false)

    const [step, setStep] = React.useState<Step>(1)
    const [loading, setLoading] = React.useState(false)
    const [error, setError] = React.useState("")

    const [paidEmail, setPaidEmail] = React.useState("")
    const [trialEmail, setTrialEmail] = React.useState("")
    const [authMessage, setAuthMessage] = React.useState("")

    const mountedRef = React.useRef(true)

    // step 1
    const [targetRoles, setTargetRoles] = React.useState("")
    const [targetLocations, setTargetLocations] = React.useState("")
    const [timeline, setTimeline] = React.useState("")
    const [hardNos, setHardNos] = React.useState("")

    // step 2
    const [currentStatus, setCurrentStatus] = React.useState("")
    const [university, setUniversity] = React.useState("")
    const [major, setMajor] = React.useState("")
    const [gradYear, setGradYear] = React.useState("")
    const [strongSkills, setStrongSkills] = React.useState("")
    const [biggestConcern, setBiggestConcern] = React.useState("")
    const [entryOpenness, setEntryOpenness] = React.useState("")

    // step 3
    const [name, setName] = React.useState("")
    const [profileText, setProfileText] = React.useState("")
    const [writingSamples, setWritingSamples] = React.useState("")
    const [constraints, setConstraints] = React.useState("")

    function setErr(msg: string) {
        setError(msg || "Something went wrong.")
        try {
            window.scrollTo({ top: 0, behavior: "smooth" })
        } catch {}
    }

    function validateStep1(): string | null {
        if (!targetRoles.trim()) {
            return "Enter at least one target role."
        }
        if (!targetLocations.trim()) {
            return "Enter at least one target location."
        }
        if (!timeline.trim()) {
            return "Pick a timeline."
        }
        return null
    }

    function validateStep2(): string | null {
        if (!currentStatus.trim()) return "Select your current status."
        if (currentStatus === "Current student") {
            if (!university.trim()) return "Enter your university."
            if (!major.trim()) return "Enter your major."
            if (!gradYear.trim()) return "Enter your graduation year."
        }
        if (!strongSkills.trim()) return "List your strongest skills."
        if (!biggestConcern.trim()) return "Tell us your biggest concern."
        if (!entryOpenness.trim()) return "Pick your openness to entry points."
        return null
    }

    function validateStep3(): string | null {
        if (!profileText.trim()) {
            return "Paste your resume or background. This is required."
        }
        if (profileText.trim().length < 300) {
            return "Your profile paste is too short. Minimum 300 characters."
        }
        return null
    }

    function goNext() {
        setError("")
        if (step === 1) {
            const e = validateStep1()
            if (e) return setErr(e)
            setStep(2)
            return
        }
        if (step === 2) {
            const e = validateStep2()
            if (e) return setErr(e)
            setStep(3)
        }
    }

    function goBack() {
        setError("")
        if (step === 2) return setStep(1)
        if (step === 3) return setStep(2)
    }

    function buildSynthesizedProfileText() {
        const parts = [
            `NAME: ${name || ""}`.trim(),
            `STATUS: ${currentStatus || ""}`.trim(),
            currentStatus === "Current student"
                ? `SCHOOL: ${university || ""} | MAJOR: ${major || ""} | GRAD: ${gradYear || ""}`
                : "",
            `TARGET ROLES: ${targetRoles || ""}`.trim(),
            `TARGET LOCATIONS: ${targetLocations || ""}`.trim(),
            `TIMELINE: ${timeline || ""}`.trim(),
            hardNos ? `HARD NOs: ${hardNos}` : "",
            `STRONG SKILLS: ${strongSkills || ""}`.trim(),
            `BIGGEST CONCERN: ${biggestConcern || ""}`.trim(),
            `ENTRY OPENNESS: ${entryOpenness || ""}`.trim(),
            constraints ? `CONSTRAINTS: ${constraints}` : "",
            "",
            "----- PASTED RESUME / BACKGROUND -----",
            profileText || "",
            writingSamples
                ? "\n----- WRITING SAMPLES (OPTIONAL) -----\n" + writingSamples
                : "",
        ]
        return parts.filter(Boolean).join("\n")
    }

    async function submitAll() {
        setError("")
        const e3 = validateStep3()
        if (e3) return setErr(e3)

        setLoading(true)
        try {
            const sessRes = await supabase.auth.getSession()
            const session = sessRes?.data?.session
            const accessToken = session?.access_token

            if (!accessToken) {
                setAuthed(false)
                throw new Error("Not signed in. Use the link from your email.")
            }

            const synthesizedProfileText = buildSynthesizedProfileText()

            const res = await fetch(`${API_BASE}/api/profile-intake`, {
                method: "POST",
                headers: {
                    "Content-Type": "application/json",
                    Authorization: `Bearer ${accessToken}`,
                },
                body: JSON.stringify({
                    name: name || null,
                    job_type: currentStatus || null,
                    target_roles: targetRoles || null,
                    target_locations: targetLocations || null,
                    preferred_locations: targetLocations || null,
                    timeline: timeline || null,
                    profile_text: synthesizedProfileText,
                    resume_text: profileText || synthesizedProfileText,
                }),
            })

            const raw = await res.text()
            let data: any = null
            try {
                data = JSON.parse(raw)
            } catch {}

            if (!res.ok) {
                const msg =
                    data?.error ||
                    data?.detail ||
                    raw ||
                    `Request failed (${res.status})`
                throw new Error(msg)
            }

            window.location.assign(JOBFIT_PATH)
        } catch (err: any) {
            setErr(err?.message || "Submission failed.")
        } finally {
            setLoading(false)
        }
    }

        mountedRef.current = true
        let alive = true

            try {
                await tryExchangeCodeForSessionIfPresent()

                const sessRes = await getSessionWithRetry()
                const session = sessRes?.data?.session
                const has = Boolean(session?.access_token)

                if (!alive || !mountedRef.current) return
                setAuthed(has)

                if (has && hasAuthStuffInUrl()) {
                    cleanAuthFromUrl()
                    window.location.replace(
                        window.location.pathname + window.location.search
                    )
                    return
                }
            } catch {
                if (!alive || !mountedRef.current) return
                setAuthed(false)
            } finally {
                if (!alive || !mountedRef.current) return
                setBooting(false)
            }
        }

        bootAuth()

        const { data: sub } = supabase.auth.onAuthStateChange(
            (_event, session) => {
                if (!alive || !mountedRef.current) return
                setAuthed(Boolean(session?.access_token))
            }
        )

        return () => {
            alive = false
            mountedRef.current = false
            sub?.subscription?.unsubscribe()
        }
    }, [])

        setError("")
        setAuthMessage("")

        const email = (paidEmail || "").trim().toLowerCase()
        if (!email) {
            setError("Enter the email tied to your paid account.")
            return

        try {
            const redirectTo =
                typeof window !== "undefined"
                    ? `${window.location.origin}${window.location.pathname}`
                    : undefined

            const { error } = await supabase.auth.signInWithOtp({
                email,
                options: { emailRedirectTo: redirectTo },
            })


            if (error) throw error

            setAuthMessage(
                "Magic link sent. Check your email and come back through that link."
            )
        } catch (err: any) {
            setError(err?.message || "Unable to send magic link.")
        }
    }

    function continueTrialAccess() {
        setError("")
        setAuthMessage("")

        const email = (trialEmail || "").trim().toLowerCase()
        if (!email) {
            setError("Enter the email tied to your trial profile.")
            return
        }

        try {
            if (typeof window !== "undefined") {
                sessionStorage.setItem("jobfit_email", email)
                localStorage.setItem("jobfit_email", email)
                localStorage.setItem("jobfit_unlocked", "true")
                window.location.assign("/signal/jobfit")
            }
        } catch (err: any) {
            setError(err?.message || "Unable to continue trial access.")
        }
    }

    if (booting) {
        return (
            <div style={page}>
                <EngineBrandMark />
                <Shell>
                    <IntakeHero />
                    <div style={headline}>Checking your access…</div>
                    <div style={subhead}>
                        Hang tight while we load your session.
                    </div>
                </Shell>
            </div>
        )
    }

    if (!authed) {
        return (
            <div style={page}>
                <EngineBrandMark />
                <Shell>
                    <IntakeHero />

                    <div style={headline}>Choose how you want to continue.</div>
                    <div style={subhead}>
                        If you already have access, enter the email tied to your
                        account below.
                    </div>

                    {error ? <div style={errorBox}>{error}</div> : null}
                    {authMessage ? (
                        <div style={infoBox}>{authMessage}</div>
                    ) : null}

                    <div style={authCardsGrid}>
                        <div style={authCard}>
                            <div style={authCardEyebrowPaid}>Full Access</div>
                            <div style={authCardTitle}>Paid account users</div>
                            <div style={authCardText}>
                                Enter your email and we’ll send you a magic link
                                to log in.
                            </div>

                            <label style={label}>Email</label>
                            <input
                                value={paidEmail}
                                onChange={(e) => setPaidEmail(e.target.value)}
                                placeholder="you@example.com"
                                style={input}
                                autoComplete="email"
                                inputMode="email"
                            />

                            <div style={{ height: 12 }} />

                            <button
                                type="button"
                                onClick={sendPaidMagicLink}
                                style={{ ...primaryBtn, width: "100%" }}
                            >
                                Send Magic Link
                            </button>
                        </div>

                        <div style={authCardTrial}>
                            <div style={authCardEyebrowTrial}>Free Trial</div>
                            <div style={authCardTitle}>Trial users</div>
                            <div style={authCardText}>
                                Enter the email tied to your saved trial profile
                                to continue.
                            </div>

                            <label style={label}>Email</label>
                            <input
                                value={trialEmail}
                                onChange={(e) => setTrialEmail(e.target.value)}
                                placeholder="you@example.com"
                                style={input}
                                autoComplete="email"
                                inputMode="email"
                            />

                            <div style={{ height: 12 }} />

                            <button
                                type="button"
                                onClick={continueTrialAccess}
                                style={{ ...secondaryBtn, width: "100%" }}
                            >
                                Continue Free Trial
                            </button>
                        </div>
                    </div>

                    <div style={upgradeBlock}>
                        <div style={upgradeTitle}>Need access?</div>

                            Purchase SIGNAL or upgrade your access here.
                        </div>

                            Purchase / Upgrade

                    <div style={finePrint}>
                        If you already clicked your magic link and still landed
                        here, open it again in the same browser you used to
                        request it.
                    </div>
                </Shell>
            </div>
        )
    }

    return (
        <div style={page}>
            <EngineBrandMark />
            <Shell>
                <IntakeHero />
                <Progress step={step} />

                {error ? <div style={errorBox}>{error}</div> : null}

                {step === 1 ? (
                    <div>
                        <div style={headline}>Where are you aiming?</div>
                        <div style={subhead}>
                            Start with the roles, locations, and timing that
                            matter most right now.
                        </div>

                        <label style={label}>Target roles *</label>
                        <input
                            value={targetRoles}
                            onChange={(e) => setTargetRoles(e.target.value)}
                            placeholder="Investment Banking Analyst, Marketing Analyst, Strategy Associate, CRE Acquisitions…"
                            style={input}
                        />

                        <div style={{ height: 12 }} />

                        <label style={label}>Target locations *</label>
                        <input
                            value={targetLocations}
                            onChange={(e) => setTargetLocations(e.target.value)}
                            placeholder="NYC (primary), Miami (secondary), ANY"
                            style={input}
                        />

                        <div style={{ height: 12 }} />

                        <label style={label}>Timeline *</label>
                        <select
                            value={timeline}
                            onChange={(e) => setTimeline(e.target.value)}
                            style={select}
                        >
                            <option value="" disabled style={optionStyle}>
                                Select one
                            </option>
                            <option value="Immediate" style={optionStyle}>
                                Immediate
                            </option>
                            <option value="Summer 2026" style={optionStyle}>
                                Summer 2026
                            </option>
                            <option value="Fall 2026" style={optionStyle}>
                                Fall 2026
                            </option>
                            <option value="Winter 2026" style={optionStyle}>
                                Winter 2026
                            </option>
                            <option value="Spring 2027" style={optionStyle}>
                                Spring 2027
                            </option>
                            <option value="Summer 2027" style={optionStyle}>
                                Summer 2027
                            </option>
                            <option value="Flexible" style={optionStyle}>
                                Flexible
                            </option>
                        </select>

                        <div style={{ height: 12 }} />

                        <label style={label}>Hard no’s</label>
                        <input
                            value={hardNos}
                            onChange={(e) => setHardNos(e.target.value)}
                            placeholder="Commission-only sales, fully remote, startups, etc."
                            style={input}
                        />

                        <div style={btnRow}>
                            <button
                                type="button"
                                onClick={goNext}
                                style={primaryBtn}
                            >
                                Continue

                {step === 2 ? (

                            Help SIGNAL understand your current situation before

                        />

                        <label style={label}>Optional: writing samples</label>

                            rows={4}

                        <div style={btnRow}>

}

// ---------- Layout components ----------
function Shell({ children }: { children: React.ReactNode }) {
    return (
        <div style={shellOuter}>
            <div style={shellInner}>{children}</div>
        </div>
    )
}

function Progress({ step }: { step: Step }) {
    return (
        <div style={{ marginBottom: 18 }}>
            <div style={progressTopRow}>
                <div style={progressLabel}>Step {step} of 3</div>
                <div style={progressHint}>
                    {step === 1
                        ? "Direction"
                        : step === 2
                          ? "Context"
                          : "Profile"}
                </div>
            </div>
            <div style={progressBar}>
                <div
                    style={{
                        ...progressFill,
                        width:
                            step === 1
                                ? "33.33%"
                                : step === 2
                                  ? "66.66%"
                                  : "100%",
                    }}
                />
            </div>
        </div>
    )
}
